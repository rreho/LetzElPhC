16void elph_q_rotate(const ELPH_cmplx* Dmats_l, const ELPH_cmplx* elph_mat_q,
17 const ELPH_cmplx* Dmats_r, const struct Lattice* lattice,
18 const bool tim_rev, ELPH_cmplx* elph_mat_Sq)
19{
20 /*
21 WARNING !! : Dmats must be for S symmetry and not for S^-1
22 * qpt is unrotated q-point in crystal coordinates
23
24 We need to compute gmn(S^-1k)*D_nn' * D^*_mm'
25
26 D_nn' = D^\dagger_nn'(S,S^-1*k) -- Eq.1
27 D_mm' = D^\dagger_mm'(S, S^-1*k + q) -Eq.2
28 elph_mat_Sq and elph_mat_q : (( nmodes, nspin, nbands, nbands))
29 Dmats = (nspin, nbands, nbands)
30 */
31 ELPH_start_clock("Elph q expand");
32
33 const ND_int nbnds = lattice->nbnds;
34 const ND_int nmodes = lattice->nmodes;
35 const ND_int nspin = lattice->nspin;
36
37 ELPH_cmplx* tmp_buffer = calloc(nbnds * nbnds, sizeof(ELPH_cmplx));
38 CHECK_ALLOC(tmp_buffer);
39
40 char g_conj = 'T';
41 if (tim_rev)
42 {
43 g_conj = 'C';
44 }
45
46 for (ND_int ispin = 0; ispin < nspin; ++ispin)
47 {
48 const ELPH_cmplx* D_r = Dmats_r + ispin * nbnds * nbnds;
49 const ELPH_cmplx* D_l = Dmats_l + ispin * nbnds * nbnds;
50
51 for (ND_int imode = 0; imode < nmodes; ++imode)
52 {
53 ND_int gshift = nbnds * nbnds * (ispin + nspin * imode);
54
55 ELPH_cmplx* g_Sq = elph_mat_Sq + gshift;
56 const ELPH_cmplx* g_q = elph_mat_q + gshift;
57
58 // 1st zero out the buffer else U.B
59 for (ND_int i = 0; i < (nbnds * nbnds); ++i)
60 {
61 g_Sq[i] = 0;
62 }
63 matmul_cmplx(g_conj, 'C', g_q, D_r, tmp_buffer, 1.0, 0.0, nbnds,
64 nbnds, nbnds, nbnds, nbnds, nbnds);
65
66 matmul_cmplx('T', 'T', tmp_buffer, D_l, g_Sq, 1.0, 0.0, nbnds,
67 nbnds, nbnds, nbnds, nbnds, nbnds);
68 }
69 }
70
71 free(tmp_buffer);
72 ELPH_stop_clock("Elph q expand");
73}
