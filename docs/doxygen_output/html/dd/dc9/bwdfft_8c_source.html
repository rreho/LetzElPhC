13void invfft3D(struct ELPH_fft_plan* plan, const ND_int nsets, ELPH_cmplx* wfcG,
14 ELPH_cmplx* wfcr, const bool conjugate)
15{
16 /*
17 Note: if conjugate is true, then the input is conjugated and then FFT is
18 performed! please note that the input wfcG is not alterned even when
19 conjugate is set to true.
20
21 a) sphere to box
22
23 b) inv FFT along Z
24
25 c) Gather Gx, Gy and scatter Nz
26
27 d) inv fft along Gx and Gy
28 */
29
30 // first some basic stuff
31 ND_int Nx = plan->fft_dims[0];
32 ND_int Ny = plan->fft_dims[1];
33 ND_int Nz = plan->fft_dims[2];
34
35 ND_int fft_buf_size = Nx * Ny * plan->nzloc;
36
37 ND_int Ny_stride = plan->nzloc * Ny;
38
39 for (ND_int iset = 0; iset < nsets; ++iset)
40 {
41 ELPH_cmplx* wfcG_tmp = wfcG + iset * plan->ngvecs_loc;
42 ELPH_cmplx* wfcr_tmp = wfcr + iset * fft_buf_size;
43
44 ND_int igvec = 0;
45 for (ND_int ixy = 0; ixy < plan->nGxyloc; ++ixy)
46 {
47 ELPH_cmplx* zfft_ptr = plan->nz_buf + ixy * Nz;
48
49 // a) sphere to box
50 for (ND_int iz = 0; iz < Nz; ++iz)
51 {
52 zfft_ptr[iz] = 0;
53 }
54
55 for (ND_int ig = 0; ig < plan->ngxy_z[ixy]; ++ig)
56 {
57 ND_int Gz = plan->gvecs[3 * igvec + 2];
58 if (Gz < 0)
59 {
60 Gz += Nz;
61 }
62 zfft_ptr[Gz] = wfcG_tmp[igvec];
63 if (conjugate)
64 {
65 zfft_ptr[Gz] = conj(zfft_ptr[Gz]);
66 }
67 ++igvec;
68 }
69 }
70
71 if (igvec != plan->ngvecs_loc)
72 {
73 error_msg("Gvec mismatch in bwd execute.");
74 }
75
76 // b) perform invfft along z
77 fftw_fun(execute)(plan->bplan_z);
78
79 // c) (i) transpose the data
80 bwd_transpose(plan); // nz_buf has (NGxy,Nz_loc)
81
82 // zero out the xy buffer and fill the fft_data buffer
83 for (ND_int ixyz = 0; ixyz < fft_buf_size; ++ixyz)
84 {
85 wfcr_tmp[ixyz] = 0;
86 }
87

89 {
90 ND_int Gx = plan->Gxy_total[2 * ixy];
91 ND_int Gy = plan->Gxy_total[2 * ixy + 1];
92 // [Gx,Gy,0] = Gx*Ny*Nzloc + Gy*Nzloc
93 ELPH_cmplx* xy_buf = wfcr_tmp + plan->nzloc * (Gy + Gx * Ny);
94 // (Nxy,Zloc)
95 memcpy(xy_buf, plan->nz_buf + ixy * plan->nzloc,
96 sizeof(ELPH_cmplx) * plan->nzloc);
97 }
98
99 // c) (ii) FFT along Y
100 for (ND_int ix = 0; ix < plan->fft_dims[0]; ++ix)
101 {
102 if (!(plan->gx_inGvecs[ix]))
103 {
104 continue;
105 }
106
107 ELPH_cmplx* wfcr_tmp_y =
108 wfcr_tmp + ix * Ny_stride; //(Gx,Ny,Nz_log)
109 ND_int iax = fftw_fun(alignment_of)((void*)wfcr_tmp_y);
110 iax /= sizeof(ELPH_cmplx);
111 fftw_fun(execute_dft)(plan->bplan_y[iax], wfcr_tmp_y, wfcr_tmp_y);
112 }
113
114 ND_int ia = fftw_fun(alignment_of)((void*)wfcr_tmp);
115 ia /= sizeof(ELPH_cmplx);
116
117 // d) FFT along X
118 fftw_fun(execute_dft)(plan->bplan_x[ia], wfcr_tmp, wfcr_tmp);
119 }
120}
