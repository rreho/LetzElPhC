14 /* This performs a transpose from

18 */

20 /*
21 Perform Alltoallv plan->nz_buf -> plan->fft_buf
22 */
23 int mpi_error;
24 int ncpus;
25 mpi_error = MPI_Comm_size(plan->comm, &ncpus);

27
28 int* gxy_counts = plan->comm_bufs;
29 int* xy_disp = gxy_counts + ncpus;
30 int* z_counts = gxy_counts + 2 * ncpus;
31 int* z_disp = gxy_counts + 3 * ncpus;
32
33 mpi_error = MPI_Alltoallv(plan->nz_buf, gxy_counts, xy_disp, ELPH_MPI_cmplx,
34 plan->fft_data, z_counts, z_disp, ELPH_MPI_cmplx,
35 plan->comm);
36 MPI_error_msg(mpi_error);
37 /* Now plan->fft_data has the data.
38 we need to redistribute and store to plan->nz_buf */
39 for (ND_int ig = 0; ig < plan->nGxyloc; ++ig)

42 for (ND_int i = 0; i < ncpus; ++i)
