16void rotate_eig_vecs(struct symmetry* sym, const struct Lattice* lattice,
17 const ELPH_float* qpt, const ELPH_cmplx* eig_q,
18 ELPH_cmplx* eig_Sq)
19{
20 // qpt is in crsytal coordinates (un rotated q point)
21 ELPH_start_clock("Phonon eig rotation");
22
23 const ELPH_float* blat = lattice->blat_vec;
24 const ND_int natom = lattice->natom;
25 const ND_int nmodes = lattice->nmodes;
26
27 // find the equalivalent positions that the symmetry maps
28 ELPH_float* atom_pos_crys = calloc(3 * natom, sizeof(ELPH_float));
29 CHECK_ALLOC(atom_pos_crys);
30
31 ND_int* rot_map = malloc(sizeof(ND_int) * natom);
32 CHECK_ALLOC(rot_map);
33
34 matmul_float('N', 'N', lattice->atomic_pos, blat, atom_pos_crys,
35 1.0 / (2 * ELPH_PI), 0.0, 3, 3, 3, natom, 3, 3);
36
37 // apply symmetry to atomic positions
38 for (ND_int ia = 0; ia < natom; ++ia)
39 {
40 const ELPH_float* pos_tmp = lattice->atomic_pos + 3 * ia;
41
42 ELPH_float rot_atom[3] = {0, 0, 0};
43 // apply rotation
44 MatVec3f(sym->Rmat, pos_tmp, false, rot_atom);
45 // add translation
46 for (int ix = 0; ix < 3; ++ix)
47 {
48 rot_atom[ix] += sym->tau[ix];
49 if (sym->time_rev)
50 {
51 rot_atom[ix] = -rot_atom[ix];
52 }
53 }
54 ELPH_float rot_atom_crys[3] = {0, 0, 0};
55 // convert to cystal
56 MatVec3f(blat, rot_atom, true, rot_atom_crys);
57 // remove 2*pi comming from blat
58 for (int ix = 0; ix < 3; ++ix)
59 {
60 rot_atom_crys[ix] /= (2 * ELPH_PI);
61 }
62 // find the index
63 // note we can use find_kidx_in_list function as the
64 // functionality is not limited to reciprocal space
65 // (contrary to what the name says so)
66 rot_map[ia] = find_kidx_in_list(natom, atom_pos_crys, rot_atom_crys);
67
68 if (rot_map[ia] < 0)
69 {
70 error_msg(
71 "Unable to find the equalivalent atom for the symmetry "
72 "operation. "
73 "Wrong symmetry operation detected.");
74 }
75 }
76 free(atom_pos_crys);
77
78 ELPH_float qcart[3];
79 ELPH_float Sq_cart[3];
80 // Sq in cart units
81
82 // convert qpt to cartisian coordinates
83 MatVec3f(blat, qpt, false, qcart); // 2pi is included
84 // compute S*q
85 MatVec3f(sym->Rmat, qcart, false, Sq_cart); // 2pi is included
86
87 ELPH_cmplx qphase = 0;
88 for (int ix = 0; ix < 3; ++ix)
89 {
90 qphase += Sq_cart[ix] * sym->tau[ix];
91 }
92 qphase = cexp(-I * qphase);
93
94 ELPH_cmplx* exp_Sqr = malloc(natom * sizeof(ELPH_cmplx));
95 // exp(I*Sq*atomic_pos)
96 CHECK_ALLOC(exp_Sqr);
97
98 ELPH_cmplx* exp_qr = malloc(natom * sizeof(ELPH_cmplx));
99 // exp(-I*q*atomic_pos)
100 CHECK_ALLOC(exp_qr);
101
102 ELPH_cmplx* rot_eig = calloc(nmodes, sizeof(ELPH_cmplx));
103 CHECK_ALLOC(rot_eig);
104
105 ELPH_cmplx Rmat_cmplx[9];
106 for (int i = 0; i < 9; ++i)
107 {
108 Rmat_cmplx[i] = sym->Rmat[i];
109 }
110
111 for (ND_int ia = 0; ia < natom; ++ia)
112 {
113 const ELPH_float* pos_tmp = lattice->atomic_pos + 3 * ia;
114 exp_Sqr[ia] = 0;
115 exp_qr[ia] = 0;
116
117 for (int ix = 0; ix < 3; ++ix)
118 {
119 exp_Sqr[ia] += Sq_cart[ix] * pos_tmp[ix];
120 exp_qr[ia] += qcart[ix] * pos_tmp[ix];
121 }
122 exp_Sqr[ia] = cexp(I * exp_Sqr[ia]);
123 exp_qr[ia] = cexp(-I * exp_qr[ia]);
124 }
125
126 for (ND_int imode = 0; imode < nmodes; ++imode)
127 {
128 ELPH_cmplx* eig_Sq_mode = eig_Sq + imode * nmodes;
129 const ELPH_cmplx* eig_q_mode = eig_q + imode * nmodes;
130
131 // S@eig_q_mode.T = (3,natom) eig_q_mode@S^T =
132 matmul_cmplx('N', 'T', eig_q_mode, Rmat_cmplx, rot_eig, 1.0, 0.0, 3, 3,
133 3, natom, 3, 3);
134
135 for (ND_int ia = 0; ia < natom; ++ia)
136 { // remove q phase
137 ELPH_cmplx* rot_eig_ia = rot_eig + ia * 3;
138 for (int ix = 0; ix < 3; ++ix)
139 {
140 rot_eig_ia[ix] *= exp_qr[ia];
141 rot_eig_ia[ix] *= qphase;
142 }
143
144 // index of atom that the symmetry maps ia atom
145 ND_int ia_rot = rot_map[ia];
146 ELPH_cmplx* eig_Sq_mode_ia = eig_Sq_mode + ia_rot * 3;
147
148 // add Sq phase back
149 for (int ix = 0; ix < 3; ++ix)
150 {
151 // in case of time reversal symmetry we need to conjugate
152 if (sym->time_rev)
153 {
154 rot_eig_ia[ix] = -conj(rot_eig_ia[ix]);
155 // the negative sign is from rotation matrix
156 }
157 eig_Sq_mode_ia[ix] = exp_Sqr[ia_rot] * rot_eig_ia[ix];
158 }
159 }
160 }
161
162 free(rot_map);
163 free(rot_eig);
164 free(exp_Sqr);
165 free(exp_qr);
166 ELPH_stop_clock("Phonon eig rotation");
167}
