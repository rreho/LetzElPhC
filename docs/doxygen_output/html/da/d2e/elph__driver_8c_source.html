69 CHECK_ALLOC(phonon);

73
74 // read the SAVE data and phonon related data.
75 read_and_alloc_save_data(input_data->save_dir, mpi_comms,
76 input_data->start_bnd, input_data->end_bnd, &wfcs,
77 input_data->ph_save_dir, lattice, pseudo, phonon,
78 dft_code);
79
80 // print info about lattice and phonons
81 print_lattice_info(mpi_comms, lattice);
82 print_phonon_info(mpi_comms, phonon);
83 //======= Now start the real computation =========
84 // a) COmpute the D_mats and store them in the netcdf file
85 // ============= Dmats =====================
86 print_info_msg(mpi_comms->commW_rank, "");
87 print_info_msg(mpi_comms->commW_rank,
88 "=== Computing Dmats for phonon symmetries ===");
89 compute_and_write_dmats("ndb.Dmats", wfcs, lattice, phonon->nph_sym,
90 phonon->ph_syms, mpi_comms);
91 // b) Compute elph
92 // ============= ELPH iBZ computation =============
93 ND_int nmodes = lattice->nmodes;
94 ND_int nfft_loc =
95 lattice->fft_dims[0] * lattice->fft_dims[1] * lattice->nfftz_loc;
96
97 ELPH_cmplx* eigVec = malloc(sizeof(ELPH_cmplx) * nmodes * nmodes);
98 //(nmodes,nmodes) // buffer to store eigen vectors
99 CHECK_ALLOC(eigVec);
100
101 ELPH_cmplx* dVscf =
102 malloc(sizeof(ELPH_cmplx) * nmodes * lattice->nmag * nfft_loc);
103 // (nmodes,nmag,Nx,Ny,Nz) // bufffer to store dVscf
104 CHECK_ALLOC(dVscf);
105
106 ELPH_float* omega_ph = malloc(sizeof(ELPH_float) * nmodes);
107 // buffer for storing phonon freq
108 CHECK_ALLOC(omega_ph);
109
110 int ncid_elph, ncid_dmat, nc_err;
111 int varid_eig, varid_elph, varid_omega, varid_dmat;
112 // Define netcdf variables
113 if (mpi_comms->commK_rank == 0)
114 {
115 // open Dmat file
116 if ((nc_err = nc_open_par("ndb.Dmats", NC_NOWRITE, mpi_comms->commR,
117 MPI_INFO_NULL, &ncid_dmat)))
118 {
119 ERR(nc_err);
120 }
121
122 // get dmat var id for dmats
123 if ((nc_err = nc_inq_varid(ncid_dmat, "Dmats", &varid_dmat)))
124 {
125 ERR(nc_err);
126 }
127
128 size_t Dmat_counts[6] = {0, 0, 0, 0, 0, 0};

130 // suitiations. This happens when a procces doesnot make atleast
131 // single read call (for ex when qpool = qiBZ)
132 if ((nc_err = nc_get_vara(ncid_dmat, varid_dmat, Dmat_counts,
133 Dmat_counts, NULL)))
134 {
135 ERR(nc_err);
136 }
137
138 // create elph file. Note: we overwrite any existing file
139 if ((nc_err =
140 nc_create_par("ndb.elph", NC_NETCDF4 | NC_CLOBBER,
141 mpi_comms->commR, MPI_INFO_NULL, &ncid_elph)))
142 {
143 fprintf(stderr, "Error creating ndb.elph file.");
144 ERR(nc_err);
145 }
146
147 // set no fill mode (to avoid writting twice)
148 if ((nc_err = ncsetfill(ncid_elph, NC_NOFILL)))
149 {
150 fprintf(stderr, "Error setting nc_fill to ndb.elph file.");
151 ERR(nc_err);
152 }
153
154 def_ncVar(ncid_elph, &varid_eig, 5, ELPH_NC4_IO_FLOAT,
155 (ND_int[]){phonon->nq_BZ, nmodes, nmodes / 3, 3, 2},
156 "POLARIZATION_VECTORS",
157 (char*[]){"nq", "nmodes", "atom", "pol", "re_im"}, NULL);
158
159 def_ncVar(ncid_elph, &varid_omega, 2, ELPH_NC4_IO_FLOAT,
160 (ND_int[]){phonon->nq_BZ, nmodes}, "FREQ",
161 (char*[]){"nq", "nmodes"}, NULL);
162
163 ND_int nk_chunk_size =
164 NC4_DEFAULT_CHUCK_KB * 1024; // now this is in bytes
165 // scale with complex number size to get the number of elements
166 nk_chunk_size /= (sizeof(ELPH_cmplx) * nmodes * lattice->nspin *
167 lattice->nbnds * lattice->nbnds);
168 // chuck the varaible elph_mat with atmost default size
169 if (nk_chunk_size == 0)
170 {
171 nk_chunk_size = 1;
172 }
173 else if (nk_chunk_size > lattice->nkpts_BZ)
174 {
175 nk_chunk_size = lattice->nkpts_BZ;
176 }
177
178 def_ncVar(ncid_elph, &varid_elph, 7, ELPH_NC4_IO_FLOAT,
179 (ND_int[]){phonon->nq_BZ, lattice->nkpts_BZ, nmodes,
180 lattice->nspin, lattice->nbnds, lattice->nbnds, 2},
181 "elph_mat",
182 (char*[]){"nq", "nk", "nmodes", "nspin", "initial_band",
183 "final_band_PH_abs", "re_im"},
184 (size_t[]){1, nk_chunk_size, nmodes, lattice->nspin,
185 lattice->nbnds, lattice->nbnds, 2});
186 }
187
188 ELPH_cmplx* eig_Sq = NULL;
189
190 if (mpi_comms->commQ_rank == 0)
191 {
192 eig_Sq = calloc(nmodes * nmodes, sizeof(ELPH_cmplx));
193 CHECK_ALLOC(eig_Sq);
194 }
195
196 print_info_msg(mpi_comms->commW_rank,
197 "=== Computing Electron-phonon matrix elements ===");
198 print_info_msg(mpi_comms->commW_rank, "");
199 for (ND_int iqpt = 0; iqpt < phonon->nq_iBZ_loc; ++iqpt)
200 {
201 print_info_msg(mpi_comms->commW_rank, "### q-point : %d/%d",
202 (int)(iqpt + 1), (int)phonon->nq_iBZ_loc);
203
204 ND_int iqpt_iBZg = iqpt + phonon->nq_shift;
205 // read dynamical matrix and dvscf for the iBZ qpt
206 if (dft_code == DFT_CODE_QE)
207 {
208 ELPH_cmplx* dvscf_read = NULL;
209 if (kernel->screening == ELPH_DFPT_SCREENING)
210 {
211 dvscf_read = dVscf;
212 }
213 else
214 {
215 // zero out the buffer. This is must !
216 ND_int dvscf_num = nmodes * lattice->nmag * nfft_loc;
217 for (ND_int ix = 0; ix < dvscf_num; ++ix)
218 {
219 dVscf[ix] = 0.0;
220 }
221 }
222 //
223 get_dvscf_dyn_qe(input_data->ph_save_dir, lattice, iqpt_iBZg,
224 eigVec, dvscf_read, omega_ph, mpi_comms);
225 // qe dvscf only contains dV_Ha + dV_xc, we need to add the local
226 // part of pseudo
227 }
228 else
229 {
230 error_msg("Currently only quantum espresso supported");
231 }
232 // local part
233 ELPH_cmplx* Vlocr = malloc(sizeof(ELPH_cmplx) * nmodes * nfft_loc);
234 // buffer to store local part of the pseudo potential
235 CHECK_ALLOC(Vlocr);
236 //
237 // compute the local part of the bare
238 dVlocq(phonon->qpts_iBZ + iqpt_iBZg * 3, lattice, pseudo, eigVec, Vlocr,
239 mpi_comms->commK);
240
241 if (dft_code == DFT_CODE_QE)
242 {
243 add_dvscf_qe(dVscf, Vlocr,
244 lattice); // add bare local to induce part
245 }
246 else
247 {
248 error_msg("Currently only quantum espresso supported");
249 }
250
251 free(Vlocr);
252 // free the local part buffer
253 ND_int qpos = 0; // positon of this iBZ qpoint in full q point list
254 for (ND_int i = 0; i < iqpt_iBZg; ++i)
255 {
256 qpos += phonon->nqstar[i];
257 }
258 // write eigen vectors and frequencies
259 if (mpi_comms->commQ_rank == 0)
260 {
261 size_t startp[5] = {qpos, 0, 0, 0, 0};
262 size_t countp[5] = {1, nmodes, nmodes / 3, 3, 2};
263 if ((nc_err =
264 nc_put_vara(ncid_elph, varid_eig, startp, countp, eigVec)))
265 {
266 ERR(nc_err);
267 }
268
269 if ((nc_err = nc_put_vara(ncid_elph, varid_omega, startp, countp,
270 omega_ph)))
271 {
272 ERR(nc_err);
273 }
274 // write down the rotate eigen vectors;
275 for (ND_int istar = 1; istar < phonon->nqstar[iqpt_iBZg]; ++istar)
276 {
277 ND_int qpos_star = qpos + istar;
278
279 struct symmetry* sym_rot =
280 phonon->ph_syms + phonon->qmap[2 * qpos_star + 1];
281
282 rotate_eig_vecs(sym_rot, lattice,
283 phonon->qpts_iBZ + iqpt_iBZg * 3, eigVec,
284 eig_Sq);
285 //
286 startp[0] = qpos_star;
287 if ((nc_err = nc_put_vara(ncid_elph, varid_eig, startp, countp,
288 eig_Sq)))
289 {
290 ERR(nc_err);
291 }
292
293 if ((nc_err = nc_put_vara(ncid_elph, varid_omega, startp,
294 countp, omega_ph)))
295 {
296 ERR(nc_err);
297 }
298 //
299 }
300 }
301 // Now compute and write the electron-phonon matrix elements
302 compute_and_write_elphq(wfcs, lattice, pseudo, phonon, iqpt_iBZg,
303 eigVec, dVscf, ncid_elph, varid_elph, ncid_dmat,
304 varid_dmat, kernel->non_loc,
305 input_data->kminusq, mpi_comms);
306 }
307
308 free(eig_Sq);
309
310 if (mpi_comms->commK_rank == 0)
311 {
312 // close files
313 if ((nc_err = nc_close(ncid_elph)))
314 {
315 ERR(nc_err);
316 }
317
318 if ((nc_err = nc_close(ncid_dmat)))
319 {
320 ERR(nc_err);
321 }
322 }
323
324 // finally write some basic info to ndb.elph file (only master node writes
325 // it)
326
327 if (mpi_comms->commW_rank == 0)
328 {
329 if ((nc_err = nc_open("ndb.elph", NC_WRITE, &ncid_elph)))
330 {
331 ERR(nc_err);
332 }
333
334 char convention_str[32];
335 strcpy(convention_str, "standard");
336 if (input_data->kminusq)
337 {
338 strcpy(convention_str, "yambo");
339 }
340 write_basic_data(ncid_elph, lattice, phonon, input_data->kernel_str,
341 convention_str);
342
343 if ((nc_err = nc_close(ncid_elph)))
344 {
345 ERR(nc_err);
346 }
347 }
348
349 int World_rank_tmp = mpi_comms->commW_rank;
350 // ELPH_cmplx Ry2Ha = pow(2,-1.5);
351 free(omega_ph);
352 free(eigVec);
353 free(dVscf);
354
355 // cleanup
356 free(kernel);
357 free_usr_input(input_data);
358 free_save_data(wfcs, lattice, pseudo, phonon);
359 free(lattice);
360 free(pseudo);
361 free(phonon);
362 free_parallel_comms(mpi_comms);
363 free(mpi_comms);
364 fftw_fun(cleanup)();
365
366 // print the clocks
367 if (0 == World_rank_tmp)
368 {
369 print_ELPH_clock_summary();
370 }
371 // cleanup the clocks
372 cleanup_ELPH_clocks();
373 // done with the calculation
374 print_info_msg(World_rank_tmp, "********** Program ended **********");
375}
