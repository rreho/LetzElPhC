15void apply_trans_wfc(const ELPH_float* trans_vec, const ELPH_float* kvec,
16 const ND_int nsets, const ND_int npw,
17 const ELPH_float* gvecs, ELPH_cmplx* wfc_G,
18 const bool conjugate)
19{
20 /*
21 This is an in-place function. wfc_G (nsets,npw) will be over written.
22 If conjugate is true, the output is conjugated
23
24 T(a) C_G * e^{i(G+k).r} = C_G * e^{i(G+k).r} * e^{-i(G+k).a}
25 note : trans_vec, kvec, gvecs can be in crystal/cart coordinates but
26 we must use the same for all. kvec and gvecs must be in 2*pi units
27 when using cart coordinates.
28 */
29
30 // first compute the phase from k i.e e^{-ik*a}
31 ELPH_cmplx kphase = cexp(-I * 2 * ELPH_PI * dot3_macro(kvec, trans_vec));
32
33 for (ND_int iset = 0; iset < nsets; ++iset)
34 {
35 ELPH_cmplx* wfc_G_tmp = wfc_G + iset * npw;
36
37 ELPH_OMP_PAR_FOR_SIMD
38 for (ND_int ig = 0; ig < npw; ++ig)
39 {
40 const ELPH_float* gvec_tmp = gvecs + 3 * ig;
41
42 ELPH_cmplx gphase = kphase * cexp(-I * 2 * ELPH_PI *
43 dot3_macro(gvec_tmp, trans_vec));
44
45 wfc_G_tmp[ig] *= gphase;
46
47 if (conjugate)
48 {
49 wfc_G_tmp[ig] = conj(wfc_G_tmp[ig]);
50 }
51 }
52 }
53}
