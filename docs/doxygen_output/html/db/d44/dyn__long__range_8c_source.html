27void add_ph_dyn_long_range(const ELPH_float* qpt, struct Lattice* lattice,
28 struct Phonon* phonon, const ND_int* Ggrid,
29 const ND_int sign, const ELPH_float* atomic_masses,
30 ELPH_cmplx* dyn_mat_asr, ELPH_cmplx* dyn_mat)
31{
32 // compute the long range part
33 add_ph_dyn_long_range_internal(qpt, lattice, phonon, Ggrid, sign,
34 atomic_masses, dyn_mat);
35 // now subtract the asr correction
36 //
37 ELPH_float factor = -1;
38 if (sign < 0)
39 {
40 factor = -factor;
41 }
42
43 ND_int nmodes = lattice->natom * 3;
44 //
45 for (ND_int ia = 0; ia < lattice->natom; ++ia)
46 {
47 ELPH_cmplx* tmp_buf_ia = dyn_mat_asr + ia * 9;
48 for (ND_int i = 0; i < 3; ++i)
49 {
50 for (ND_int j = 0; j < 3; ++j)
51 {
52 dyn_mat[(ia * 3 + i) * nmodes + ia * 3 + j] +=
53 factor * tmp_buf_ia[3 * i + j];
54 }
55 }
56 }
57}

59void compute_dyn_lr_asr_correction(struct Lattice* lattice,
60 struct Phonon* phonon, const ND_int* Ggrid,
61 const ELPH_float* atomic_masses,
62 ELPH_cmplx* dyn_mat_asr)
63{
64 // ASR for the lr non-analytical dynamical matrix.
65 // (dyn_mat_asr) : (natom, 3, 3)
66 // See X. Gonze PhysRevB.50.13035
67 //
68 for (ND_int i = 0; i < lattice->natom * 9; ++i)
69 {
70 dyn_mat_asr[i] = 0.0;
71 }
72
73 if (!phonon->epsilon || !phonon->Zborn)
74 {
75 return;
76 }
77 ND_int nmodes = 3 * lattice->natom;
78 ELPH_cmplx* tmp_dyn_mat = calloc(nmodes * nmodes, sizeof(*tmp_dyn_mat));
79 CHECK_ALLOC(tmp_dyn_mat);
80 // let the comiplers remove this loop.
81 for (ND_int i = 0; i < nmodes * nmodes; ++i)
82 {
83 tmp_dyn_mat[i] = 0.0;
84 }
85
86 ELPH_float qpt_zero[3] = {0.0, 0.0, 0.0};
87 add_ph_dyn_long_range_internal(qpt_zero, lattice, phonon, Ggrid, 1,
88 atomic_masses, tmp_dyn_mat);
89
90 //
91 for (ND_int ia = 0; ia < lattice->natom; ++ia)
92 {
93 ELPH_cmplx* tmp_buf_ia = dyn_mat_asr + ia * 9;
94 for (ND_int ja = 0; ja < lattice->natom; ++ja)
95 {
96 for (ND_int i = 0; i < 3; ++i)
97 {
98 for (ND_int j = 0; j < 3; ++j)
99 {
100 tmp_buf_ia[3 * i + j] +=
101 tmp_dyn_mat[(ia * 3 + i) * nmodes + ja * 3 + j];
102 }
103 }
104 }
105 }
106 free(tmp_dyn_mat);
107}

109static void add_ph_dyn_long_range_internal(
110 const ELPH_float* qpt, struct Lattice* lattice, struct Phonon* phonon,
111 const ND_int* Ggrid, const ND_int sign, const ELPH_float* atomic_masses,
112 ELPH_cmplx* dyn_mat)
113{
114 // adds or subtracts non-analytical term to the dynamical matrix.
115 // sign < 0 : subtract else add
116 // qpt in crystal coordinater
117 //
118 // (3D) X. Gonze et al Phys. Rev. B 50, 13035(R)
119 // (2D ) T. Sohier et al Nano Lett. 2017, 17, 6, 3758–3763
120 // (2D) S. Ponce et al PHYSICAL REVIEW B 107, 155424 (2023)
121 //
122 if (!phonon->epsilon || (!phonon->Zborn && !phonon->Qpole))
123 {
124 return;
125 }
126 //
127 ELPH_start_clock("dyn lr part");
128
129 if (lattice->dimension == '2' && fabs(qpt[2]) > ELPH_EPS)
130 {
131 error_msg(
132 "In 2D, only qz == 0 points are accepted when interpolating.");
133 }
134
135 ELPH_cmplx factor = 4.0 * ELPH_PI * ELPH_e2 / lattice->volume; // prefactor
136 if (sign < 0)
137 {
138 factor = -factor;
139 }
140 //
141 ELPH_float eps_alpha[9];
142
143 memcpy(eps_alpha, phonon->epsilon, sizeof(eps_alpha));
144
145 ELPH_float zlat = lattice->alat_vec[8];
146 //
147 if (lattice->dimension == '2')
148 {
149 factor = factor * zlat / 2.0;
150 // compute alpha = c/2 * (eps-1)
151 for (int i = 0; i < 9; ++i)
152 {
153 eps_alpha[i] = 0.0;
154 }
155 // inplane
156 eps_alpha[0] = 0.5 * zlat * (phonon->epsilon[0] - 1);
157 eps_alpha[1] = 0.5 * zlat * (phonon->epsilon[1]);
158 eps_alpha[3] = 0.5 * zlat * (phonon->epsilon[3]);
159 eps_alpha[4] = 0.5 * zlat * (phonon->epsilon[4] - 1);
160 }
161
162 ND_int GridZ = Ggrid[2];
163 if (lattice->dimension == '2')
164 {
165 GridZ = 1;
166 }
167 //
168 ND_int Gvec_size = Ggrid[0] * Ggrid[1] * GridZ;
169
170 ND_int natom = lattice->natom;
171
172 ELPH_cmplx* Zdotq_tau =
173 calloc(2 * Gvec_size * 3 * natom, sizeof(*Zdotq_tau));
174 CHECK_ALLOC(Zdotq_tau);
175 // leave it to the compilers(most will very likely remove it )
176 for (ND_int i = 0; i < (2 * Gvec_size * 3 * natom); ++i)
177 {
178 Zdotq_tau[i] = 0.0;
179 }
180 //
181 ELPH_OMP_PAR_FOR_SIMD
182 for (ND_int ig = 0; ig < Gvec_size; ++ig)
183 {
184 ELPH_cmplx* out_tmp1 = Zdotq_tau + ig * 3 * natom;
185 ELPH_cmplx* out_tmp2 =
186 Zdotq_tau + ig * 3 * natom + Gvec_size * 3 * natom;
187 const ND_int Gx = ig / (Ggrid[1] * GridZ);
188 const ND_int Gy = (ig % (Ggrid[1] * GridZ)) / GridZ;
189 const ND_int Gz = (ig % (Ggrid[1] * GridZ)) % GridZ;
190 //
191 ELPH_float qplusG[3], tmp_buf[3];
192 tmp_buf[0] = (qpt[0] + get_miller_idx(Gx, Ggrid[0]));
193 tmp_buf[1] = (qpt[1] + get_miller_idx(Gy, Ggrid[1]));
194 tmp_buf[2] = (qpt[2] + get_miller_idx(Gz, GridZ));
195 // COnvert to cart units (2*pi) is included
196 MatVec3f(lattice->blat_vec, tmp_buf, false, qplusG);
197 //
198 ELPH_float qplusG_norm = sqrt(dot3_macro(qplusG, qplusG));
199 if (qplusG_norm < ELPH_EPS)
200 {
201 continue;
202 }
203 //
204 MatVec3f(eps_alpha, qplusG, false, tmp_buf);
205 ELPH_float q_eps_q = dot3_macro(tmp_buf, qplusG);
206
207 ELPH_float decay_fac = exp(-q_eps_q * q_eps_q * 0.125);
208
209 if (lattice->dimension == '2')
210 {
211 q_eps_q += qplusG_norm;
212 decay_fac = exp(-qplusG_norm * qplusG_norm * 0.125);
213 }
214 //
215 for (ND_int ia = 0; ia < natom; ++ia)
216 {
217 const ELPH_float* Z_k =
218 phonon->Zborn ? (phonon->Zborn + 9 * ia) : NULL;
219 const ELPH_float* Q_k =
220 phonon->Qpole ? (phonon->Qpole + 27 * ia) : NULL;
221 const ELPH_float* tau_k = lattice->atomic_pos + 3 * ia;
222
223 ELPH_float Qpole_buf[3] = {0.0, 0.0, 0.0};
224 //
225 if (Z_k)
226 {
227 MatVec3f(Z_k, qplusG, true, tmp_buf);
228 }
229 else
230 {
231 memcpy(tmp_buf, Qpole_buf, sizeof(tmp_buf));
232 }
233 //
234 ELPH_cmplx* out_tmp_buf = out_tmp1 + 3 * ia;
235 ELPH_cmplx* out_tmp_buf2 = out_tmp2 + 3 * ia;
236
237 // compute (q+G)_x Q_xyz * (q+G)_y
238 if (Q_k)
239 {
240 for (int i = 0; i < 3; ++i)
241 {
242 for (int j = 0; j < 3; ++j)
243 {
244 for (int k = 0; k < 3; ++k)
245 {
246 Qpole_buf[k] =
247 Qpole_buf[k] +
248 qplusG[i] * qplusG[j] * Q_k[k + 3 * j + 9 * i];
249 }
250 }
251 }
252 if (lattice->dimension == '2')
253 {
254 // subtract Qzz|q+G|^2
255 for (int i = 0; i < 3; ++i)
256 {
257 Qpole_buf[i] -=
258 (qplusG_norm * qplusG_norm * Q_k[i + 24]);
259 }
260 }
261 }
262 // e^{-iq.tau}
263 ELPH_cmplx qdot_tau = cexp(-I * (dot3_macro(qplusG, tau_k)));
264 qdot_tau /= q_eps_q;
265 qdot_tau *= decay_fac;
266 qdot_tau /= sqrt(atomic_masses[ia]);
267 //
268 for (ND_int i = 0; i < 3; ++i)
269 {
270 out_tmp_buf[i] =
271 (tmp_buf[i] - I * 0.5 * Qpole_buf[i]) * qdot_tau;
272 out_tmp_buf2[i] = out_tmp_buf[i] * q_eps_q;
273 }
274 }
275 }
276 //
277 ND_int nmodes = 3 * natom;
278 matmul_cmplx('C', 'N', Zdotq_tau + Gvec_size * 3 * natom, Zdotq_tau,
279 dyn_mat, factor, 1.0, nmodes, nmodes, nmodes, nmodes, nmodes,
280 Gvec_size);
281 //
282 free(Zdotq_tau);
283 //
284 //
285 ELPH_stop_clock("dyn lr part");
286 //
287 return;
288}
