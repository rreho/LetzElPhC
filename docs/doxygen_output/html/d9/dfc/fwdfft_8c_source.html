13void fft3D(struct ELPH_fft_plan* plan, const ND_int nsets, ELPH_cmplx* wfcr,
14 ELPH_cmplx* wfcG, const bool conjugate)
15{
16 /*
17 Note: if conjugate is true, then the output of FFT is conjugated
18
19 a) FFT along xy and (Nx,Ny,Nz_loc)-> (Gx, Gy, Nz_loc)
20
21 b) scatter Gx,Gy to different cpus to perform fft along x and y.
22 *Note that we use scatter instead of alltoallv so that some FFTs
23 can be computed while being communicating
24
25 c) Perform FFT along Z
26
27 d) box to sphere
28 */
29
30 ND_int Nx = plan->fft_dims[0];
31 ND_int Ny = plan->fft_dims[1];
32 ND_int Nz = plan->fft_dims[2];
33
34 ELPH_float norm = Nx * Ny * Nz;
35 norm = 1.0 / norm;
36
37 ND_int Ny_stride = plan->nzloc * Ny;
38
39 ND_int fft_buf_size = Nx * Ny * plan->nzloc;
40
41 for (ND_int iset = 0; iset < nsets; ++iset)
42 {
43 ELPH_cmplx* wfcr_tmp = wfcr + iset * fft_buf_size;
44 ELPH_cmplx* wfcG_tmp = wfcG + iset * plan->ngvecs_loc;
45
46 ND_int ia = fftw_fun(alignment_of)((void*)wfcr_tmp);
47 ia /= sizeof(ELPH_cmplx);
48 // a) (i) FFT along X
49 fftw_fun(execute_dft)(plan->fplan_x[ia], wfcr_tmp, wfcr_tmp);
50
51 // a) (ii) FFT along Y
52 for (ND_int ix = 0; ix < plan->fft_dims[0]; ++ix)
53 {
54 if (!(plan->gx_inGvecs[ix]))
55 {
56 continue;
57 }
58
59 ELPH_cmplx* wfcr_tmp_y =
60 wfcr_tmp + ix * Ny_stride; //(Gx,Ny,Nz_log)
61 ND_int iax = fftw_fun(alignment_of)((void*)wfcr_tmp_y);
62 iax /= sizeof(ELPH_cmplx);
63 fftw_fun(execute_dft)(plan->fplan_y[iax], wfcr_tmp_y, wfcr_tmp_y);
64 }
65 // b) (i) pack the data for transpose
66 for (ND_int ixy = 0; ixy < plan->nGxy; ++ixy)
67 {
68 ND_int Gx = plan->Gxy_total[2 * ixy];
69 ND_int Gy = plan->Gxy_total[2 * ixy + 1];
70 // [Gx,Gy,0] = Gx*Ny*Nzloc + Gy*Nzloc
71 ELPH_cmplx* xy_buf = wfcr_tmp + plan->nzloc * (Gy + Gx * Ny);
72 // (Nxy,Zloc)
73 memcpy(plan->nz_buf + ixy * plan->nzloc, xy_buf,
74 sizeof(ELPH_cmplx) * plan->nzloc);
75 }
76 // b) (ii) transpose the data
77 fwd_transpose(plan); // nz_buf has (NGxy_loc,Nz)
78
79 // c) perform fft along z
80 fftw_fun(execute)(plan->fplan_z);
81
82 // d) box to sphere
83 ND_int igvec = 0;
84 for (ND_int ixy = 0; ixy < plan->nGxyloc; ++ixy)
85 {
86 ELPH_cmplx* zfft_ptr = plan->nz_buf + ixy * Nz;
87 for (ND_int ig = 0; ig < plan->ngxy_z[ixy]; ++ig)
88 {
89 ND_int Gz = plan->gvecs[3 * igvec + 2];
90 if (Gz < 0)
91 {
92 Gz += Nz;
93 }
94 wfcG_tmp[igvec] = zfft_ptr[Gz] * norm;
95 if (conjugate)
96 {
97 wfcG_tmp[igvec] = conj(wfcG_tmp[igvec]);
98 }
99 ++igvec;
100 }
101 }
102 if (igvec != plan->ngvecs_loc)
103 {
104 error_msg("Gvec mismatch in fwd execute. ");
105 }
106 }
