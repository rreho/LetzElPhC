14ND_int get_mpi_local_size_idx(const ND_int n, ND_int* start_idx, MPI_Comm Comm)
15{
16 int my_rank, total_size;
17 int mpi_error;
18 mpi_error = MPI_Comm_rank(Comm, &my_rank);
19 MPI_error_msg(mpi_error);
20
21 mpi_error = MPI_Comm_size(Comm, &total_size);
22 MPI_error_msg(mpi_error);
23
24 ND_int n_q = n / total_size;
25 ND_int n_r = n % total_size;
26
27 ND_int n_this_cpu = n_q;
28 if (my_rank < n_r)
29 {
30 ++n_this_cpu;
31 }
32
33 if (start_idx != NULL)
34 {
35 *start_idx = n_q * my_rank;
36 if (my_rank < n_r)
37 {
38 *start_idx += my_rank;
39 }
40 else
41 {
42 *start_idx += n_r;
43 }
44 }
45 return n_this_cpu;
46}

49 ND_int* shift)
50{
51 /*
52 This function is used to distribute k/q points between n groups
53 return k/q for each group, and the shift
54 */
55
56 ND_int n_q = n / ngrp;
57 ND_int n_r = n % ngrp;
58
59 ND_int n_this_grp = n_q;
60 if (igrp < n_r)
61 {
62 ++n_this_grp;
63 }
64
65 *shift = n_q * igrp;
66 if (igrp < n_r)
67 {
68 *shift += igrp;
69 }
70 else
71 {
72 *shift += n_r;
73 }
74

79void create_parallel_comms(const int nqpools, const int nkpools,
80 const MPI_Comm MPI_world_comm,
81 struct ELPH_MPI_Comms* Comm)
82{
83 /*
84 !! Warning : Order of arrangement of cpus is very important. the key given
85 to comm_split must be the rank of world comm
86 */
87 int mpi_error;
88 // first set the basic things
89 Comm->commW = MPI_world_comm;
90 Comm->nqpools = nqpools;
91 Comm->nkpools = nkpools;
92
93 mpi_error = MPI_Comm_rank(Comm->commW, &Comm->commW_rank);
94 MPI_error_msg(mpi_error);
95
96 mpi_error = MPI_Comm_size(Comm->commW, &Comm->commW_size);
97 MPI_error_msg(mpi_error);
98
99 // check Comm->commW_size is divisible by nqpools*nkpools
100 if (nqpools < 1 || nkpools < 1)
101 {
102 error_msg(" number of pools must be greater than 0 ");
103 }
104 if (Comm->commW_size % (nqpools * nkpools) != 0)
105 {
106 error_msg(" product of kpools and qpools must divide total cpus.");
107 }
108
109 Comm->commQ_size = Comm->commW_size / nqpools;
110 Comm->commK_size = Comm->commQ_size / nkpools;
111
112 /* create comms */
113 // commQ
114 mpi_error = MPI_Comm_split(Comm->commW, Comm->commW_rank / Comm->commQ_size,
115 Comm->commW_rank, &Comm->commQ);
116 MPI_error_msg(mpi_error);
117
118 mpi_error = MPI_Comm_rank(Comm->commQ, &Comm->commQ_rank);
119 MPI_error_msg(mpi_error);
120
121 // commK
122 mpi_error = MPI_Comm_split(Comm->commQ, Comm->commQ_rank / Comm->commK_size,
123 Comm->commW_rank, &Comm->commK);
124 MPI_error_msg(mpi_error);
125
126 mpi_error = MPI_Comm_rank(Comm->commK, &Comm->commK_rank);
127 MPI_error_msg(mpi_error);
128 // commR
129 mpi_error = MPI_Comm_split(Comm->commW, Comm->commK_rank, Comm->commW_rank,
130 &Comm->commR);
131 MPI_error_msg(mpi_error);
132
133 mpi_error = MPI_Comm_rank(Comm->commR, &Comm->commR_rank);
134 MPI_error_msg(mpi_error);
135
136 mpi_error = MPI_Comm_size(Comm->commR, &Comm->commR_size);
137 MPI_error_msg(mpi_error);
138 // commRq
139 mpi_error = MPI_Comm_split(Comm->commQ, Comm->commK_rank, Comm->commW_rank,
140 &Comm->commRq);
141 MPI_error_msg(mpi_error);
142
143 mpi_error = MPI_Comm_rank(Comm->commRq, &Comm->commRq_rank);
144 MPI_error_msg(mpi_error);
145
146 mpi_error = MPI_Comm_size(Comm->commRq, &Comm->commRq_size);
147 MPI_error_msg(mpi_error);
148
149 // Sanity check
150 if (Comm->commK_rank != Comm->commW_rank % Comm->commK_size)
151 {
152 error_msg("MPI Comm 1 creation failed.");
153 }
154 if (Comm->commR_size != (nqpools * nkpools))
155 {
156 error_msg("MPI Comm 2 creation failed.");
157 }
158 if (Comm->commRq_size != nkpools)
159 {
160 error_msg("MPI Comm 3 creation failed.");
161 }
162}
