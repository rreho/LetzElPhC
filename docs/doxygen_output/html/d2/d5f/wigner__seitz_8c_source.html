28ND_int build_wigner_seitz_vectors(const ND_int *grid,
29 const ELPH_float *lat_vecs, double eps,
30 const ELPH_float *rvec_m, ND_int nrvec_m,
31 const ELPH_float *rvec_n, ND_int nrvec_n,
32 ND_int **ws_vecs, ND_int **ws_degen)
33{
34 // Note. you need to free buffer allocated outside of this functions
35 // find all vectors T such that |r_m-(r_n+R+T)|(or |(r_n+R+T)-r_m|) is
36 // is mim for a given grid, R vecs are miller indices given by
37 // "get_miller_idx" function "common/numerical_func.c"
38 // R+T are the wigner seitz vectors for particular atomic position.
39 //
40 // Eq. 46 of G Pizzi et al 2020 J. Phys.: Condens. Matter 32 165902
41 //
42 // NM : we assume that order of R vecs is (X,Y,Z) in row major i.e
43 // Rz variying fast followed by Ry and then Rx
44 //
45 // rvec_m and rvec_n are in cart units
46 // must free ws_vecs and ws_degen outside of the function
47 // ws_degen has a size = grid[0]*grid[1]*grid[2]*nrvec_n*nrvec_m with
48 // ws_degen[i] (Rx,Ry,Rz,nrvec_m, nrvec_n) give the degeneracy of R point.
49 // ws_vecs has as size = 3*sum(ws_degen)
50 // return sum(ws_degen)
51 //
52 ELPH_start_clock("Wigner seitz");
53
54 struct kdtree *tree =
55 setup_ws_tree(grid, lat_vecs, WS_SUPERCELL_SEARCH_SIZE);
56 if (!tree)
57 {
58 error_msg("Buildling KD Tree for wigner seitz cell failed.");
59 }
60
61 ELPH_float blat[9];
62 reciprocal_vecs(lat_vecs, blat);
63 // remove 2*pi
64 for (ND_int xi = 0; xi < 9; ++xi)
65 {
66 blat[xi] /= (2 * ELPH_PI);
67 }
68
69 const ELPH_float zerovec[3] = {0.0, 0.0, 0.0};
70 // create a tmp buffer to read points.
71 size_t tree_count = 2 * WS_SUPERCELL_SEARCH_SIZE + 1;
72 tree_count = tree_count * tree_count * tree_count;
73 ELPH_float *pts_buf = malloc(3 * tree_count * sizeof(*pts_buf));
74 CHECK_ALLOC(pts_buf);
75 //
76 //
77 if (!rvec_m)
78 {
79 nrvec_m = 1;
80 rvec_m = zerovec;
81 }
82 if (!rvec_n)
83 {
84 nrvec_n = 1;
85 rvec_n = zerovec;
86 }
87 //
88 const ND_int nRpts = grid[0] * grid[1] * grid[2];
89 const ND_int nRmnpts = nRpts * nrvec_m * nrvec_n;
90 //
91 // intially allocated 2*nRmnpts ws vectors. if more
92 // needed, we can realloc.
93 ND_int ws_vec_size = nRmnpts * 2;
94
95 ND_int *ws_vec_buf = malloc(3 * ws_vec_size * sizeof(*ws_vec_buf));
96 CHECK_ALLOC(ws_vec_buf);
97
98 ND_int *ws_degen_buf = calloc(nRmnpts, sizeof(*ws_degen_buf));
99 CHECK_ALLOC(ws_degen_buf);
100 *ws_degen = ws_degen_buf;
101 //
102 ND_int nws_vec_found = 0;
103 //
104 ND_int Gridyz = grid[1] * grid[2];
105 //
106 for (ND_int i = 0; i < nRpts; ++i)
107 {
108 ND_int Rx = i / Gridyz;
109 ND_int Ry = (i % Gridyz) / grid[2];
110 ND_int Rz = (i % Gridyz) % grid[2];
111 //
112 Rx = get_miller_idx(Rx, grid[0]);
113 Ry = get_miller_idx(Ry, grid[1]);
114 Rz = get_miller_idx(Rz, grid[2]);
115 //
116 ELPH_float Rvec[3];
117 Rvec[0] = lat_vecs[0] * Rx + lat_vecs[1] * Ry + lat_vecs[2] * Rz;
118 Rvec[1] = lat_vecs[3] * Rx + lat_vecs[4] * Ry + lat_vecs[5] * Rz;
119 Rvec[2] = lat_vecs[6] * Rx + lat_vecs[7] * Ry + lat_vecs[8] * Rz;
120 //
121 for (ND_int im = 0; im < nrvec_m; ++im)
122 {
123 for (ND_int in = 0; in < nrvec_n; ++in)
124 {
125 ND_int ishift = in + im * nrvec_n + i * nrvec_m * nrvec_n;
126 double query_pnt[3];
127 query_pnt[0] =
128 rvec_m[3 * im + 0] - rvec_n[3 * in + 0] - Rvec[0];
129 query_pnt[1] =
130 rvec_m[3 * im + 1] - rvec_n[3 * in + 1] - Rvec[1];
131 query_pnt[2] =
132 rvec_m[3 * im + 2] - rvec_n[3 * in + 2] - Rvec[2];
133 // find the norm of query point
134 double query_norm2 = sqrt(dot3_macro(query_pnt, query_pnt));
135 //
136 ND_int i_ws_found = get_ws_nearest_superlat(
137 tree, query_pnt, eps * query_norm2 + 1e-8, pts_buf);
138 if (0 == i_ws_found)
139 {
140 error_msg("No Wigner Seitz vector found.");
141 }
142 //
143 ws_degen_buf[ishift] = i_ws_found;
144 if (nws_vec_found + i_ws_found > ws_vec_size)
145 {
146 // realloc.
147 ws_vec_size = ws_vec_size + (MAX(nRmnpts, i_ws_found));
148 ND_int *realloc_ptr = realloc(
149 ws_vec_buf, 3 * ws_vec_size * sizeof(*ws_vec_buf));
150 if (!realloc_ptr)
151 {
152 free(ws_vec_buf);
153 }
154 CHECK_ALLOC(realloc_ptr);
155 ws_vec_buf = realloc_ptr;
156 }
157 // Copy the wigner seitz vectors.
158 for (ND_int ii = 0; ii < i_ws_found; ++ii)
159 {
160 //
161 ELPH_float tmp_out[3];
162 tmp_out[0] = blat[0] * pts_buf[3 * ii] +
163 blat[3] * pts_buf[3 * ii + 1] +
164 blat[6] * pts_buf[3 * ii + 2];
165 tmp_out[1] = blat[1] * pts_buf[3 * ii] +
166 blat[4] * pts_buf[3 * ii + 1] +
167 blat[7] * pts_buf[3 * ii + 2];
168 tmp_out[2] = blat[2] * pts_buf[3 * ii] +
169 blat[5] * pts_buf[3 * ii + 1] +
170 blat[8] * pts_buf[3 * ii + 2];
171 //
172 ws_vec_buf[3 * nws_vec_found] = rint(tmp_out[0]);
173 ws_vec_buf[3 * nws_vec_found + 1] = rint(tmp_out[1]);
174 ws_vec_buf[3 * nws_vec_found + 2] = rint(tmp_out[2]);
175 // Do a sanity check
176 tmp_out[0] -= ws_vec_buf[3 * nws_vec_found];
177 tmp_out[1] -= ws_vec_buf[3 * nws_vec_found + 1];
178 tmp_out[2] -= ws_vec_buf[3 * nws_vec_found + 2];
179 if (fabs(tmp_out[0]) > 1e-3 || fabs(tmp_out[1]) > 1e-3 ||
180 fabs(tmp_out[2]) > 1e-3)
181 {
182 error_msg(
183 "Something wrong with kdtree. not a lattice "
184 "vector.");
185 }
186 //
187 ++nws_vec_found;
188 }
189 }
190 }
191 }
192
193 // shrink the buffer
194 if (nws_vec_found)
195 {
196 ND_int *realloc_ptr =
197 realloc(ws_vec_buf, 3 * nws_vec_found * sizeof(*ws_vec_buf));
198 if (realloc_ptr)
199 {
200 ws_vec_buf = realloc_ptr;
201 }
202 }
203 // set the ws_vec pointer
204 *ws_vecs = ws_vec_buf;
205 //
206
207 free(pts_buf);
208 kd_free(tree);
209 //
210 ELPH_stop_clock("Wigner seitz");
211 return nws_vec_found;
212}

216static struct kdtree *setup_ws_tree(const ND_int *grid,
217 const ELPH_float *lat_vecs,
218 const ND_int ws_ssize)
219{
220 // build a kdree for superlattice search
221 struct kdtree *tree = kd_create(3);
222 if (!tree)
223 {
224 return tree;
225 }
226 //
227 ELPH_float superlat_vecs[9];
228 // a[:,i] are latvecs
229 for (ND_int i = 0; i < 3; ++i)
230 {
231 for (ND_int j = 0; j < 3; ++j)
232 {
233 superlat_vecs[j * 3 + i] = lat_vecs[j * 3 + i] * grid[i];
234 }
235 }
236 //
237 for (ND_int i = -ws_ssize; i < ws_ssize + 1; ++i)
238 {
239 for (ND_int j = -ws_ssize; j < ws_ssize + 1; ++j)
240 {
241 for (ND_int k = -ws_ssize; k < ws_ssize + 1; ++k)
242 {
243 double treepoint[3];
244 treepoint[0] = superlat_vecs[0] * i + superlat_vecs[1] * j +
245 superlat_vecs[2] * k;
246 treepoint[1] = superlat_vecs[3] * i + superlat_vecs[4] * j +
247 superlat_vecs[5] * k;
248 treepoint[2] = superlat_vecs[6] * i + superlat_vecs[7] * j +
249 superlat_vecs[8] * k;
250 //
251 kd_insert(tree, treepoint, NULL);
252 }
253 }
254 }
255 //
256 return tree;
257}
258static ND_int get_ws_nearest_superlat(struct kdtree *tree, double *query_pnt,
259 const double eps, ELPH_float *pts_buf)
260{
261 // here we assume that the user gives
262 // enough buffer to the function.
263 // the size is atmost 3*get_ws_tree_size()
264 // return number of point found
265 // epsilon is the thresould
266 // The return points will be in cart units
267 if (!tree)
268 {
269 return 0;
270 }
271 // first get the nearest neighbour.
272 struct kdres *nearest = kd_nearest(tree, query_pnt);
273 if (!nearest)
274 {
275 error_msg("Nearest neighbor search failed");
276 }
277 //
278 // Get the distance and position of the nearest neighbor
279 double nearest_pos[3];
280 kd_res_item(nearest, nearest_pos);
281 kd_res_free(nearest);
282 //
283 nearest_pos[0] -= query_pnt[0];
284 nearest_pos[1] -= query_pnt[1];
285 nearest_pos[2] -= query_pnt[2];
286 //
287 double nearest_dist = sqrt(dot3_macro(nearest_pos, nearest_pos)) + 1e-10;
288 // NM : we add a small absoulte tolerance so to protect eps being 0
289
290 // now find all within radius
291 struct kdres *range =
292 kd_nearest_range(tree, query_pnt, nearest_dist + fabs(eps));
293 if (!range)
294 {
295 error_msg("Cannot find wigner seitz vectors.");
296 }
297 //
298 kd_res_rewind(range);
299 ND_int Nfound = 0;
300 while (!kd_res_end(range))
301 {
302 double pos[3];
303 kd_res_item(range, pos);
304 pts_buf[3 * Nfound] = pos[0];
305 pts_buf[3 * Nfound + 1] = pos[1];
306 pts_buf[3 * Nfound + 2] = pos[2];
307 kd_res_next(range);
308 ++Nfound;
309 }
310 kd_res_free(range);
311
312 return Nfound;
313}
