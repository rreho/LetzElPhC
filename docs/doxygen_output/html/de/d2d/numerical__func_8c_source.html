27/* Function to compute legendre transforms */
28ELPH_float legendre(int l_val, int m_val, ELPH_float x_in)
29{
30 /* This does not contain Condon–Shortley phase
31 Eq: 9-12 in
32 "Miguel A. Blanco, M. Flórez, M. Bermejo
33 Evaluation of the rotation matrices in the basis of real spherical harmonics
34 J. Mol. Struct. THEOCHEM 419, 19-27 (1997)"
35 */
36 /* Order of below if-else conditions is important */
37 if (l_val == 0 && m_val == 0)
38 {
39 return 1.0;
40 }
41
42 else if (l_val < 0)
43 {
44 return legendre(-l_val - 1, m_val, x_in);
45 }
46
47 else if (abs(m_val) > l_val)
48 {
49 return 0.0;
50 }
51
52 else if (m_val < 0)
53 {
54 return legendre(l_val, -m_val, x_in) * pow(-1, -m_val) *
55 factorial(l_val + m_val) / factorial(l_val - m_val);
56 }
57 else
58 {
59 ELPH_float legendre_val = 1.0;
60 ELPH_float sqrt_xin = sqrt(1 - (x_in * x_in));
61
62 /* P^l_l = (2l-1)*sqrt(1-x^2)*P^{l-1}_{l-1}(x) */
63 for (int im = 1; im <= m_val; ++im)
64 {
65 legendre_val *= (2 * im - 1) * sqrt_xin;
66 }
67
68 if (l_val == m_val)
69 {
70 return legendre_val;
71 }
72
73 ELPH_float legendre_val_mm = legendre_val;
74 /* P^l_l+1 = (2l+1)*x*P^l_l*/
75 legendre_val *= (2 * m_val + 1) * x_in;
76
77 if (l_val == (m_val + 1))
78 {
79 return legendre_val;
80 }
81
82 /* For all other l,m combination */
83 for (int im = m_val + 2; im <= l_val; ++im)
84 {
85 ELPH_float legendre_temp = legendre_val;
86 legendre_val = (2 * im - 1) * x_in * legendre_val -
87 (im + m_val - 1) * legendre_val_mm;
88 legendre_val = legendre_val / (im - m_val);
89 legendre_val_mm = legendre_temp;
90 }
91 return legendre_val;
92 }
93}

95/* Compute real spherical Harmonics for vector direction*/
96ELPH_float Ylm(int l_val, int m_val, ELPH_float* vec)
97{
98 /*
99 Miguel A. Blanco, M. Flórez, M. Bermejo
100 Evaluation of the rotation matrices in the basis of real spherical harmonics
101 J. Mol. Struct. THEOCHEM 419, 19-27 (1997)
102 */
103 if (l_val < 0)
104 {
105 return 0.0; // error
106 }
107
108 ELPH_float cost, phi;
109 ELPH_float norm = sqrt(vec[0] * vec[0] + vec[1] * vec[1] + vec[2] * vec[2]);
110
111 if (norm < ELPH_EPS)
112 {
113 if (l_val != 0)
114 {
115 return 0;
116 }
117 else
118 {
119 return 1.0 / sqrt(4 * ELPH_PI);
120 }
121 }
122
123 cost = vec[2] / norm; // z/r
124 phi = atan2(vec[1], vec[0]);
125
126 ELPH_float sh = 0.0;
127
128 /* phi part*/
129 if (m_val > 0)
130 {
131 sh = sqrt(2) * cos(m_val * phi);
132 }
133 else if (m_val < 0)
134 {
135 m_val = -m_val;
136 sh = sqrt(2) * sin(m_val * phi);
137 }
138 else
139 {
140 sh = 1; // 1 for m =0
141 }
142
143 /* Multiply with legendre polynomial */
144 sh *= legendre(l_val, m_val, cost);
145
146 /* Multiply with normalization factor*/
147 sh *= sqrt(factorial(l_val - m_val) * (2 * l_val + 1) /
148 (4 * ELPH_PI * factorial(l_val + m_val)));
149
150 return sh;
151}

153/* Function for simpson integration*/
154ELPH_float simpson(const ELPH_float* func_vals, const ELPH_float* dx,
155 ND_int npts)
156{
157 /*
158 Compute the integral using simpson 1/3 rules i.e /int f(x) dx
159 */
160 if (npts < 3)
161 {
162 return 0; // Need atleast 3 points . Return an error instead
163 }
164
165 ELPH_float sum = func_vals[0] * dx[0];
166
167 /* Odd terms*/
168 for (ND_int i = 1; i < npts - 1; i += 2)
169 {
170 sum += 4.0 * func_vals[i] * dx[i];
171 }
172
173 /* even terms*/
174 for (ND_int i = 2; i < npts - 1; i += 2)
175 {
176 sum += 2.0 * func_vals[i] * dx[i];
177 }
178
179 if (npts % 2 != 0)
180 {
181 sum += func_vals[npts - 1] * dx[npts - 1];
182 }
183 else
184 {
185 sum -= func_vals[npts - 3] * dx[npts - 3] * 0.25;
186 sum += func_vals[npts - 2] * dx[npts - 2];
187 sum += func_vals[npts - 1] * dx[npts - 1] * 1.25;
188 }
189
190 return sum / 3.0;
191}

193/* Function to compute cosine of angle between two vectors*/
194ELPH_float cos_angle_bw_Vec(const ELPH_float* vec1, const ELPH_float* vec2)
195{
196 /*
197 This function returns Cos(w) where w is angle between vec1 and vec2
198 */
199 ELPH_float norm1 =
200 vec1[0] * vec1[0] + vec1[1] * vec1[1] + vec1[2] * vec1[2];
201 ELPH_float norm2 =
202 vec2[0] * vec2[0] + vec2[1] * vec2[1] + vec2[2] * vec2[2];
203 ELPH_float norm = sqrt(norm1 * norm2);
204 if (norm < ELPH_EPS)
205 {
206 return 0;
207 }
208 ELPH_float dot_12 =
209 vec1[0] * vec2[0] + vec1[1] * vec2[1] + vec1[2] * vec2[2];
210 return dot_12 / norm;
211}

270ELPH_float det3x3(const ELPH_float* mat)
271{
272 ELPH_float det = 0;
273 det += mat[0] * (mat[4] * mat[8] - mat[5] * mat[7]);
274 det -= mat[1] * (mat[3] * mat[8] - mat[5] * mat[6]);
275 det += mat[2] * (mat[7] * mat[3] - mat[4] * mat[6]);
276 return det;

279void reciprocal_vecs(const ELPH_float* lat_vec, ELPH_float* restrict blat)
280{
281 /*
282 a[:,i] are latvecs
283 b[:,i] are blat
284 result is multiplied with 2*pi
285 */
286 ELPH_float det = det3x3(lat_vec);
287 if (det < ELPH_EPS)
288 {
289 error_msg("Inverting singular matrix");
290 }
291 det = 2 * ELPH_PI / det;
292 blat[0] = (lat_vec[4] * lat_vec[8] - lat_vec[7] * lat_vec[5]) * det;
293 blat[1] = (lat_vec[5] * lat_vec[6] - lat_vec[3] * lat_vec[8]) * det;
294 blat[2] = (lat_vec[3] * lat_vec[7] - lat_vec[6] * lat_vec[4]) * det;
295 blat[3] = (lat_vec[2] * lat_vec[7] - lat_vec[1] * lat_vec[8]) * det;
296 blat[4] = (lat_vec[0] * lat_vec[8] - lat_vec[2] * lat_vec[6]) * det;
297 blat[5] = (lat_vec[1] * lat_vec[6] - lat_vec[0] * lat_vec[7]) * det;
298 blat[6] = (lat_vec[1] * lat_vec[5] - lat_vec[2] * lat_vec[4]) * det;
299 blat[7] = (lat_vec[2] * lat_vec[3] - lat_vec[0] * lat_vec[5]) * det;
300 blat[8] = (lat_vec[0] * lat_vec[4] - lat_vec[1] * lat_vec[3]) * det;
301}

435ND_int get_miller_idx(ND_int idx_in, ND_int FFT_dimension)
436{
437 // returns FFT Indices to [-N/2, N/2) if FFT_dimension is even
438 // [-(n-1)/2,(n-1)/2 )] of FFT_dimension is odd
439 ND_int mid_pnt = (FFT_dimension - 1) / 2 + 1;
440 if (idx_in < mid_pnt)
441 {
442 return idx_in;
443 }
444 else
445 {
446 return (idx_in - mid_pnt) - FFT_dimension / 2;
447 }

470ND_int find_kidx_in_list(ND_int nkpts, const ELPH_float* kpts_list,
471 const ELPH_float* kpt)
472{
473 /*
474 kpt and kpts_list must be in crystal coordinates
475 return -1 if not found
476 */
477 ND_int kidx = -1;
478 for (ND_int ik = 0; ik < nkpts; ++ik)
479 {
480 const ELPH_float* ik_vec_tmp = kpts_list + 3 * ik;
481 ELPH_float sum = 0;
482 for (int i = 0; i < 3; ++i)
483 {
484 ELPH_float diff_tmp = ik_vec_tmp[i] - kpt[i];
485 diff_tmp = diff_tmp - rint(diff_tmp);
486 sum += diff_tmp * diff_tmp;
487 }
488 sum = sqrt(sum);
489 if (sum < ELPH_EPS)
490 {
491 kidx = ik;
492 break;
493 }
494 }
495 return kidx;
496}

499void get_KplusQ_idxs(const ND_int Nbz, const ELPH_float* kpoints,
500 const ELPH_float* Q_pt, int* KplusQidxs)

508
