116/*
117 * Compute the greatest common divisor of a and b.
118 */
119static int gcd(int a, int b)
120{
121 int c;
122
123 c = a % b;
124 while (c != 0)
125 {
126 a = b;
127 b = c;
128 c = a % b;
129 }
130
131 return (b);
132}

134/*
135 * Exchange the block from nonopt_start to nonopt_end with the block
136 * from nonopt_end to opt_end (keeping the same order of arguments
137 * in each block).
138 */
139static void permute_args(int panonopt_start, int panonopt_end, int opt_end,
140 char* const* nargv)
141{
142 int cstart, cyclelen, i, j, ncycle, nnonopts, nopts, pos;
143 char* swap;
144
145 /*
146 * compute lengths of blocks and number and size of cycles
147 */
148 nnonopts = panonopt_end - panonopt_start;
149 nopts = opt_end - panonopt_end;
150 ncycle = gcd(nnonopts, nopts);
151 cyclelen = (opt_end - panonopt_start) / ncycle;
152
153 for (i = 0; i < ncycle; i++)
154 {
155 cstart = panonopt_end + i;
156 pos = cstart;
157 for (j = 0; j < cyclelen; j++)
158 {
159 if (pos >= panonopt_end)
160 {
161 pos -= nnonopts;
162 }
163 else
164 {
165 pos += nopts;
166 }
167 swap = nargv[pos];
168 /* LINTED const cast */
169 ((char**)nargv)[pos] = nargv[cstart];
170 /* LINTED const cast */
171 ((char**)nargv)[cstart] = swap;
172 }
173 }
174}

176/*
177 * parse_long_options --
178 * Parse long options in argc/argv argument vector.
179 * Returns -1 if short_too is set and the option does not match long_options.
180 */
181static int parse_long_options(char* const* nargv, const char* options,
182 const struct option* long_options, int* idx,
183 int short_too, int flags)
184{
185 char *current_argv, *has_equal;
186#ifdef GNU_COMPATIBLE
187 char* current_dash;
188#endif
189 size_t current_argv_len;
190 int i, match, exact_match, second_partial_match;
191
192 current_argv = place;
193#ifdef GNU_COMPATIBLE
194 switch (dash_prefix)
195 {
196 case D_PREFIX:
197 current_dash = "-";
198 break;
199 case DD_PREFIX:
200 current_dash = "--";
201 break;
202 case W_PREFIX:
203 current_dash = "-W ";
204 break;
205 default:
206 current_dash = "";
207 break;
208 }
209#endif
210 match = -1;
211 exact_match = 0;
212 second_partial_match = 0;
213
214 optind++;
215
216 if ((has_equal = strchr(current_argv, '=')) != NULL)
217 {
218 /* argument found (--option=arg) */
219 current_argv_len = has_equal - current_argv;
220 has_equal++;
221 }
222 else
223 {
224 current_argv_len = strlen(current_argv);
225 }
226
227 for (i = 0; long_options[i].name; i++)
228 {
229 /* find matching long option */
230 if (strncmp(current_argv, long_options[i].name, current_argv_len))
231 {
232 continue;
233 }
234
235 if (strlen(long_options[i].name) == current_argv_len)
236 {
237 /* exact match */
238 match = i;
239 exact_match = 1;
240 break;
241 }
242 /*
243 * If this is a known short option, don't allow
244 * a partial match of a single character.
245 */
246 if (short_too && current_argv_len == 1)
247 {
248 continue;
249 }
250
251 if (match == -1) /* first partial match */
252 {
253 match = i;
254 }
255 else if ((flags & FLAG_LONGONLY) ||
256 long_options[i].has_arg != long_options[match].has_arg ||
257 long_options[i].flag != long_options[match].flag ||
258 long_options[i].val != long_options[match].val)
259 {
260 second_partial_match = 1;
261 }
262 }
263 if (!exact_match && second_partial_match)
264 {
265 /* ambiguous abbreviation */
266 if (PRINT_ERROR)
267 {
268 fprintf(stderr, ambig,
269#ifdef GNU_COMPATIBLE
270 current_dash,
271#endif
272 (int)current_argv_len, current_argv);
273 }
274 optopt = 0;
275 return (BADCH);
276 }
277 if (match != -1)
278 { /* option found */
279 if (long_options[match].has_arg == no_argument && has_equal)
280 {
281 if (PRINT_ERROR)
282 {
283 fprintf(stderr, noarg,
284#ifdef GNU_COMPATIBLE
285 current_dash,
286#endif
287 (int)current_argv_len, current_argv);
288 }
289 /*
290 * XXX: GNU sets optopt to val regardless of flag
291 */
292 if (long_options[match].flag == NULL)
293 {
294 optopt = long_options[match].val;
295 }
296 else
297 {
298 optopt = 0;
299 }
300#ifdef GNU_COMPATIBLE
301 return (BADCH);
302#else
303 return (BADARG);
304#endif
305 }
306 if (long_options[match].has_arg == required_argument ||
307 long_options[match].has_arg == optional_argument)
308 {
309 if (has_equal)
310 {
311 optarg = has_equal;
312 }
313 else if (long_options[match].has_arg == required_argument)
314 {
315 /*
316 * optional argument doesn't use next nargv
317 */
318 optarg = nargv[optind++];
319 }
320 }
321 if ((long_options[match].has_arg == required_argument) &&
322 (optarg == NULL))
323 {
324 /*
325 * Missing argument; leading ':' indicates no error
326 * should be generated.
327 */
328 if (PRINT_ERROR)
329 {
330 fprintf(stderr, recargstring,
331#ifdef GNU_COMPATIBLE
332 current_dash,
333#endif
334 current_argv);
335 }
336 /*
337 * XXX: GNU sets optopt to val regardless of flag
338 */
339 if (long_options[match].flag == NULL)
340 {
341 optopt = long_options[match].val;
342 }
343 else
344 {
345 optopt = 0;
346 }
347 --optind;
348 return (BADARG);
349 }
350 }
351 else
352 { /* unknown option */
353 if (short_too)
354 {
355 --optind;
356 return (-1);
357 }
358 if (PRINT_ERROR)
359 {
360 fprintf(stderr, illoptstring,
361#ifdef GNU_COMPATIBLE
362 current_dash,
363#endif
364 current_argv);
365 }
366 optopt = 0;
367 return (BADCH);
368 }
369 if (idx)
370 {
371 *idx = match;
372 }
373 if (long_options[match].flag)
374 {
375 *long_options[match].flag = long_options[match].val;
376 return (0);
377 }
378 else
379 {
380 return (long_options[match].val);
381 }
382}

384/*
385 * getopt_internal --
386 * Parse argc/argv argument vector. Called by user level routines.
387 */
388static int getopt_internal(int nargc, char* const* nargv, const char* options,
389 const struct option* long_options, int* idx,
390 int flags)
391{
392 char* oli; /* option letter list index */
393 int optchar, short_too;
394 int posixly_correct; /* no static, can be changed on the fly */
395
396 if (options == NULL)
397 {
398 return (-1);
399 }
400
401 /*
402 * Disable GNU extensions if POSIXLY_CORRECT is set or options
403 * string begins with a '+'.
404 */
405 posixly_correct = (getenv("POSIXLY_CORRECT") != NULL);
406#ifdef GNU_COMPATIBLE
407 if (*options == '-')
408 {
409 flags |= FLAG_ALLARGS;
410 }
411 else if (posixly_correct || *options == '+')
412 {
413 flags &= ~FLAG_PERMUTE;
414 }
415#else
416 if (posixly_correct || *options == '+')
417 {
418 flags &= ~FLAG_PERMUTE;
419 }
420 else if (*options == '-')
421 {
422 flags |= FLAG_ALLARGS;
423 }
424#endif
425 if (*options == '+' || *options == '-')
426 {
427 options++;
428 }
429
430 /*
431 * XXX Some GNU programs (like cvs) set optind to 0 instead of
432 * XXX using optreset. Work around this braindamage.
433 */
434 if (optind == 0)
435 {
436 optind = optreset = 1;
437 }
438
439 optarg = NULL;
440 if (optreset)
441 {
442 nonopt_start = nonopt_end = -1;
443 }
444start:
445 if (optreset || !*place)
446 { /* update scanning pointer */
447 optreset = 0;
448 if (optind >= nargc)
449 { /* end of argument vector */
450 place = EMSG;
451 if (nonopt_end != -1)
452 {
453 /* do permutation, if we have to */
454 permute_args(nonopt_start, nonopt_end, optind, nargv);
455 optind -= nonopt_end - nonopt_start;
456 }
457 else if (nonopt_start != -1)
458 {
459 /*
460 * If we skipped non-options, set optind
461 * to the first of them.
462 */
463 optind = nonopt_start;
464 }
465 nonopt_start = nonopt_end = -1;
466 return (-1);
467 }
468 if (*(place = nargv[optind]) != '-' ||
469#ifdef GNU_COMPATIBLE
470 place[1] == '\0')
471 {
472#else
473 (place[1] == '\0' && strchr(options, '-') == NULL))
474 {
475#endif
476 place = EMSG; /* found non-option */
477 if (flags & FLAG_ALLARGS)
478 {
479 /*
480 * GNU extension:
481 * return non-option as argument to option 1
482 */
483 optarg = nargv[optind++];
484 return (INORDER);
485 }
486 if (!(flags & FLAG_PERMUTE))
487 {
488 /*
489 * If no permutation wanted, stop parsing
490 * at first non-option.
491 */
492 return (-1);
493 }
494 /* do permutation */
495 if (nonopt_start == -1)
496 {
497 nonopt_start = optind;
498 }
499 else if (nonopt_end != -1)
500 {
501 permute_args(nonopt_start, nonopt_end, optind, nargv);
502 nonopt_start = optind - (nonopt_end - nonopt_start);
503 nonopt_end = -1;
504 }
505 optind++;
506 /* process next argument */
507 goto start;
508 }
509 if (nonopt_start != -1 && nonopt_end == -1)
510 {
511 nonopt_end = optind;
512 }
513
514 /*
515 * If we have "-" do nothing, if "--" we are done.
516 */
517 if (place[1] != '\0' && *++place == '-' && place[1] == '\0')
518 {
519 optind++;
520 place = EMSG;
521 /*
522 * We found an option (--), so if we skipped
523 * non-options, we have to permute.
524 */
525 if (nonopt_end != -1)
526 {
527 permute_args(nonopt_start, nonopt_end, optind, nargv);
528 optind -= nonopt_end - nonopt_start;
529 }
530 nonopt_start = nonopt_end = -1;
531 return (-1);
532 }
533 }
534
535 /*
536 * Check long options if:
537 * 1) we were passed some
538 * 2) the arg is not just "-"
539 * 3) either the arg starts with -- we are getopt_long_only()
540 */
541 if (long_options != NULL && place != nargv[optind] &&
542 (*place == '-' || (flags & FLAG_LONGONLY)))
543 {
544 short_too = 0;
545#ifdef GNU_COMPATIBLE
546 dash_prefix = D_PREFIX;
547#endif
548 if (*place == '-')
549 {
550 place++; /* --foo long option */
551#ifdef GNU_COMPATIBLE
552 dash_prefix = DD_PREFIX;
553#endif
554 }
555 else if (*place != ':' && strchr(options, *place) != NULL)
556 {
557 short_too = 1; /* could be short option too */
558 }
559
560 optchar = parse_long_options(nargv, options, long_options, idx,
561 short_too, flags);
562 if (optchar != -1)
563 {
564 place = EMSG;
565 return (optchar);
566 }
567 }
568
569 if ((optchar = (int)*place++) == (int)':' ||
570 (optchar == (int)'-' && *place != '\0') ||
571 (oli = strchr(options, optchar)) == NULL)
572 {
573 /*
574 * If the user specified "-" and '-' isn't listed in
575 * options, return -1 (non-option) as per POSIX.
576 * Otherwise, it is an unknown option character (or ':').
577 */
578 if (optchar == (int)'-' && *place == '\0')
579 {
580 return (-1);
581 }
582 if (!*place)
583 {
584 ++optind;
585 }
586#ifdef GNU_COMPATIBLE
587 if (PRINT_ERROR)
588 {
589 fprintf(stderr, posixly_correct ? illoptchar : gnuoptchar, optchar);
590 }
591#else
592 if (PRINT_ERROR)
593 {
594 fprintf(stderr, illoptchar, optchar);
595 }
596#endif
597 optopt = optchar;
598 return (BADCH);
599 }
600 if (long_options != NULL && optchar == 'W' && oli[1] == ';')
601 {
602 /* -W long-option */
603 if (*place) /* no space */
604 /* NOTHING */;
605 else if (++optind >= nargc)
606 { /* no arg */
607 place = EMSG;
608 if (PRINT_ERROR)
609 {
610 fprintf(stderr, recargchar, optchar);
611 }
612 optopt = optchar;
613 return (BADARG);
614 }
615 else /* white space */
616 {
617 place = nargv[optind];
618 }
619#ifdef GNU_COMPATIBLE
620 dash_prefix = W_PREFIX;
621#endif
622 optchar =
623 parse_long_options(nargv, options, long_options, idx, 0, flags);
624 place = EMSG;
625 return (optchar);
626 }
627 if (*++oli != ':')
628 { /* doesn't take argument */
629 if (!*place)
630 {
631 ++optind;
632 }
633 }
634 else
635 { /* takes (optional) argument */
636 optarg = NULL;
637 if (*place) /* no white space */
638 {
639 optarg = place;
640 }
641 else if (oli[1] != ':')
642 { /* arg not optional */
643 if (++optind >= nargc)
644 { /* no arg */
645 place = EMSG;
646 if (PRINT_ERROR)
647 {
648 fprintf(stderr, recargchar, optchar);
649 }
650 optopt = optchar;
651 return (BADARG);
652 }
653 else
654 {
655 optarg = nargv[optind];
656 }
657 }
658 place = EMSG;
659 ++optind;
660 }
661 /* dump back option letter */
662 return (optchar);
663}
