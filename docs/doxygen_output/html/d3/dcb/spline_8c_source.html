13ELPH_float spline_interpolate(const ELPH_float x, ND_int inear,
14 const ELPH_float* xi, const ELPH_float* yi,
15 const ELPH_float* dy)
16{
17 ELPH_float tx = (x - xi[inear]) / (xi[inear + 1] - xi[inear]);
18 ELPH_float ax =
19 dy[inear] * (xi[inear + 1] - xi[inear]) - (yi[inear + 1] - yi[inear]);
20 ELPH_float bx = -dy[inear + 1] * (xi[inear + 1] - xi[inear]) +
21 (yi[inear + 1] - yi[inear]);
22 return (1 - tx) * yi[inear] + tx * yi[inear + 1] +
23 tx * (1 - tx) * ((1 - tx) * ax + tx * bx);
24}

88 ELPH_float si = (yin[i + 1] - yin[i]) / dxi;
89 ELPH_float si_n = (yin[i] - yin[i - 1]) / dxi_n;
