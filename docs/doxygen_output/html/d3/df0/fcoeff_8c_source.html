24void alloc_and_Compute_f_Coeff(struct Lattice* lattice, struct Pseudo* pseudo)
25{
26 /*
27 This functions allocates memory and computes f coefficients for spinors
28 which is given by f^{l,j,\sigma,\sigma'}_{m,m'} =\sum_{m_j}
29 P^{l,j,\sigma}_{m_j,m} * conj(P^{l,j,\sigma'}_{m_j,m'})
30
31 where P^{l,j,\sigma}_{m_j,m} = \alpha^{\sigma,l,j}_{m_j} *
32 U^{\sigma,l,j}_{m_j,m} Refer Eq.9 of PHYSICAL REVIEW B 71, 115106 s2005d
33
34 Input : PP_table from yambo. This contains l,j values (l*j,ntype,3)
35 (l+1,2j,pp_spin), pp_spin = 1 ? Output : (l*j,ntype) array. each element of
36 array has a shape (2l+1, 2l+1, nspinor, nspinor)
37
38 Note, we only compute these matrices once.
39
40 also return no of Coeffs allocated.
41
42 Note: donot forget to free the memory
43 */
44
45 int nspinor = lattice->nspinor;
46
47 if (nspinor == 1 || !lattice->is_soc_present)
48 {
49 pseudo->fCoeff = NULL;
50 return;
51 }
52
53 ND_int nltimesj = pseudo->nltimesj;
54 ND_int ntype = pseudo->ntype; // number of atomic types
55 ND_int nproj = ntype * nltimesj;
56
57 ELPH_float* PP_table = pseudo->PP_table;
58 pseudo->fCoeff = malloc(sizeof(ELPH_cmplx*) * nproj);
59 CHECK_ALLOC(pseudo->fCoeff);
60 // (ntype,l*j)
61 ELPH_cmplx** Coeff = pseudo->fCoeff;
62
63 // initiate to NULL (this is must else U.B when freeing)
64 for (ND_int i = 0; i < nproj; ++i)
65 {
66 Coeff[i] = NULL;
67 }
68
69 for (ND_int i = 0; i < nproj; ++i)
70 {
71 ND_int lidx = i / ntype;
72 ND_int itype = i % ntype;
73 // (l*j,ntype,3) [ntype*3,3,1]
74 int l = rint(PP_table[ntype * 3 * lidx + itype * 3] -
75 1); // PP_table[lidx,itype,0]
76 int j = rint(
77 PP_table[ntype * 3 * lidx + itype * 3 + 1]); // Careful, this is 2j
78
79 if (l < 0 || j == 0)
80 {
81 continue;
82 } // skip fake entries or j ==0 (j =0 if one of the pseudo does not
83 // have soc)
84
85 ND_int two_lp1 = 2 * l + 1;
86 ND_int coeff_len = two_lp1 * two_lp1 * nspinor * nspinor;
87 Coeff[i] = malloc(sizeof(ELPH_cmplx) *
88 coeff_len); //(two_lp1,two_lp1,nspinor,nspinor)
89 CHECK_ALLOC(Coeff[i]);
90 // zero out the buffer
91 for (ND_int inum = 0; inum < coeff_len; ++inum)
92 {
93 Coeff[i][inum] = 0;
94 }
95
96 bool jp1 = true; // This is true when j = l + 1/2 else false
97
98 if (j == 2 * l + 1)
99 {
100 jp1 = true;
101 }
102 else if (j == 2 * l - 1)
103 {
104 jp1 = false;
105 }
106 else
107 {
108 error_msg("Inconsistant j value in Pseudo potential.");
109 }
110
111 for (ND_int il1 = 0; il1 < 2 * l + 1; ++il1) // m
112 {
113 for (ND_int il2 = 0; il2 < 2 * l + 1; ++il2) // m'
114 {
115 for (ND_int is1 = 0; is1 < nspinor; ++is1) // sigma
116 {
117 for (ND_int is2 = 0; is2 < nspinor; ++is2) // sigma'
118 {
119 int im1 = il1 - l;
120 int im2 = il2 - l;
121 /* is =0 for spin up and is = 1 for spin down
122 so we can use "is" as boolean for spin down
123 */
124 ELPH_cmplx sum = 0;
125
126 for (int mj = -j; mj <= j; mj += 2)
127 {
128 sum += CGCoeff(jp1, is1, l, mj) *
129 CGCoeff(jp1, is2, l, mj) *
130 Umat(jp1, is1, l, mj, im1) *
131 conj(Umat(jp1, is2, l, mj, im2));
132 }
133
134 Coeff[i][nspinor * nspinor * (il1 * two_lp1 + il2) +
135 is1 * nspinor + is2] = sum;
136 // Coeff[i][il1,il2,is1,is2] = sum
137 }
138 }
139 }
140 }
141 }
142 return;
143}

166static ELPH_cmplx Umat(bool jp1, bool spin_down, int l, int twomj, int m2)
167{
168 /* These are the U matrices entering into Eq.6 of PHYSICAL REVIEW B 71,
169 * 115106 s2005 */
170 if (abs(m2) > l)
171 {
172 error_msg("m value greater than l");
173 }
174 if (jp1)
175 {
176 int m = (twomj - 1) / 2;
177 if (!spin_down)
178 {
179 return Cmat(m, m2);
180 }
181 else
182 {
183 return Cmat(m + 1, m2);
184 }
185 }
186 else
187 {
188 int m = (twomj + 1) / 2;
189 if (!spin_down)
190 {
191 return Cmat(m - 1, m2);
192 }
193 else
194 {
195 return Cmat(m, m2);
196 }
197 }
198 return 0.0;
199}

201/* Function to compute the transformation matrix from real to complex spherical
202 * harmonics. */
203static ELPH_cmplx Cmat(int m1, int m2)
204{
205 /*
206 This function returns transformation coeffients for transforming
207 Real to complex spherical harmonics. These are C^\dagger elements given in
208 Eq. 19 of Ref:
209 Miguel A. Blanco, M. Flórez, M. Bermejo
210 Evaluation of the rotation matrices in the basis of real spherical harmonics
211 J. Mol. Struct. THEOCHEM 419, 19-27 (1997)
212
213 It should be noted that definition of these elements are implementation
214 specific.
215 */
216 int abs_m1 = abs(m1);
217 int abs_m2 = abs(m2);
218
219 if (abs_m1 != abs_m2)
220 {
221 return 0.0;
222 }
223 if (m1 == 0 && m2 == 0)
224 {
225 return 1.0;
226 }
227 if (m1 == abs_m1 && m2 == abs_m2)
228 {
229 return pow(-1.0, abs_m1) / ELPH_SQRT2;
230 }
231 if (m1 == -abs_m1 && m2 == abs_m2)
232 {
233 return 1.0 / ELPH_SQRT2;
234 }
235 if (m1 == abs_m1 && m2 == -abs_m2)
236 {
237 return I * pow(-1.0, abs_m1) / ELPH_SQRT2;
238 }
239 if (m1 == -abs_m1 && m2 == -abs_m2)
240 {
241 return -I / ELPH_SQRT2;
242 }
243 else
244 {
245 error_msg("Incorrect m1 and m2 values");
246 }
247 return 0;
248}

250static ELPH_float CGCoeff(bool jp1, bool spin_down, int l, int twomj)
251{ /*
252 Function to compute Clebsch-Gordan coefficients alpha
253 jp1 is true for j = l+1/2 else false.
254 Eq-5 of Ref : PHYSICAL REVIEW B 71, 115106 s2005d
255 */
256 /* check if integrand is non negative */
257 /* Warning twomj = 2*j, we pass 2j instead of j*/
258 ELPH_float mj = twomj / 2.0;
259 if (jp1)
260 {
261 ELPH_float m = mj - 0.5;
262 if (!spin_down)
263 {
264 return sqrt((l + m + 1.0) / (2.0 * l + 1.0));
265 }
266 else
267 {
268 return sqrt((l - m) / (2.0 * l + 1.0));
269 }
270 }
271 else
272 {
273 ELPH_float m = mj + 0.5;
274 if (!spin_down)
275 {
276 return sqrt((l - m + 1.0) / (2.0 * l + 1.0));
277 }
278 else
279 {
280 return -sqrt((l + m) / (2.0 * l + 1.0));
281 }
282 }
