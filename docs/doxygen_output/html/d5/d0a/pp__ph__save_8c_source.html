470static ND_int find_nqpools(const char* out_dir, char* buffer_tmp,
471 ND_int buffer_size)
472{
473 // find number of qpools used in ph.x
474 ND_int counter = 0;
475
476 char ph0_tmp[16];
477
478 while (true)
479 {
480 snprintf(ph0_tmp, sizeof(ph0_tmp) - 1, "_ph%d", (int)counter);
481 cwk_path_join_multiple((const char*[]){out_dir, ph0_tmp, NULL},
482 buffer_tmp, buffer_size);
483 if (0 != check_dir_exists(buffer_tmp))
484 {
485 // issue with dir (either cannot access or does not exist)
486 break;
487 }
488 ++counter;
489 }
