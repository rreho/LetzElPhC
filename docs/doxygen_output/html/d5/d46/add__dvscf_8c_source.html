15 /*
16 nffts = Nx*Ny*Nz_loc
17
18 This function adds dVloc is added to dVscf.
19 dVscf_{2x2} = d{V}*I + d{Bx}*sigma_x + d{By}*sigma_y + dBz}*sigma_z
20 = | d{V} + d{Bz} d{Bx} - I*d{By} |
21 | d{Bx}+ I*d{By} d{V} - d{Bz} |
22 where Bx,By,Bz = d{Exc}/dm and V = V_Ha + d{Exc}/dn i.e V = V_Ha + V_Xc
23
24 In QE. dvscf is stored as follows :
25 nmag = 1 : dV is stored (non magnetic calc with both collinear and non
26 collinear) nmag = 2 : d{V} + d{Bz} and d{V} - d{Bz} are stored (nspin = 2)
27 nmag = 4 : d{V}, d{Bx},d{By},d{Bz} are stored (magnetic, non collinear)
28
29 dVscf -> (nmodes, nmag, nffts)
30 dVloc -> (nmodes, nffts)
31 */
32 const ND_int nmag = lattice->nmag;
33 const ND_int nmodes = lattice->nmodes;
34 const ND_int nffts =
35 lattice->fft_dims[0] * lattice->fft_dims[1] * lattice->nfftz_loc;
36
37 ND_int mag_iter = 1;
38
39 if (nmag == 2)

41 mag_iter = 2;
42 }
43 for (ND_int iv = 0; iv < nmodes; ++iv)

45 ELPH_cmplx* dVscf_mode = dVscf + iv * nffts * nmag;
46 const ELPH_cmplx* dVbare = dVloc + iv * nffts;

48 /* add local bare part of pseudo potential (i,e dVloc) to V_Ha + V_xc */
49 for (ND_int im = 0; im < mag_iter; ++im)

52 ELPH_OMP_PAR_FOR_SIMD

62 | V + Bz Bx - I*By |
63 | Bx+ I*By V - Bz |
64 where Bx,By,Bz = d{Exc}/dm and Vxc = d{Exc}/dn

66

73 ELPH_cmplx* By_loc = dVscf_mode + 2 * nffts;
74 ELPH_cmplx* Bz_loc = dVscf_mode + 3 * nffts;
75
76 ELPH_OMP_PAR_FOR_SIMD
77 for (ND_int i = 0; i < nffts; ++i)
78 {
79 ELPH_cmplx Vxc_t, Bx_t, By_t, Bz_t;
80 Vxc_t = Vxc_loc[i];
81 Bx_t = Bx_loc[i];
82 By_t = By_loc[i];
83 Bz_t = Bz_loc[i];
84
85 Vxc_loc[i] = Vxc_t + Bz_t;
86 Bx_loc[i] = Bx_t - I * By_t;
87 By_loc[i] = Bx_t + I * By_t;
88 Bz_loc[i] = Vxc_t - Bz_t;
89 }
90 }
